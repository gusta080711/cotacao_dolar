# cotacao_dolar
pseudocodigo cotação dolar

declaracoes:

cotacao-decimal;
dolar-inteiro;
resultado-decimal.


	inicio;

	escreva("digite a cotação do dolar");
		leia(cotacao);
	escreva("digite um valor em doláres para converção");
		leia(dolar);
	
	resultado<-dolar*cotacao.

	se(dolar<10)entao;
		escreva("o valor digitado deve ser maior que 10");

	fim-se.

	escreva("o resultado é");

fim-algoritimo.
